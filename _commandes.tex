\newcounter{int}
\newcommand{\reponse}[1]{
  \setcounter{int}{-1}
  \loop
  \stepcounter{int}
  \ifnum \value{int} < #1
  \lignePointillee{0.99} \\[8pt]
  \repeat
  \ifnum #1 > 0
    \vspace*{-12pt}
  \fi
}
